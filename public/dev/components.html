  <div class="modal-box">
    <h3 class="text-lg font-bold">Title</h3>
    <p class="py-4">Content here.</p>
    <div class="modal-action">
      <form method="dialog">

      </form>
      <button class="btn btn-primary">Confirm</button>
    </div>
  </div>
